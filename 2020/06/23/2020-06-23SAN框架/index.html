SAN框架

2020-06-23
<script>
    // 1.定义SAN
    const MyApp = san.defineComponent({
        template: `
            <div>
                <input type="text" value="{=name=}">
                <p>Hello {{name}}!</p>
            </div>
        `,
        // 初始化值
        initData(){
            return {
                name: "San"
            }
        },
        // SAN与服务器进行操作时，在生命周期inited() attached()进行操作
        attached() {
            console.log(this);
        }
    });
    // 2.实例化
    let myApp = new MyApp();
    // 3.attach到页面
    myApp.attach(document.body);
</script>
